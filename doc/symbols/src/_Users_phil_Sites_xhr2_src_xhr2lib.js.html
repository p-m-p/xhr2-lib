 58 	/**
 59 		Serialises a data object into a query string
 60 
 61 		@public
 62 		@param {String} url the destination url
 63 		@param {Object} data the request data
 64 	*/
 65 	this.serializeQS = function (url, data) {
 66 		var qs = "";
 67 		if (typeof data === "object") {
 68 			if (!~url.search(/\?/)) {
 69 				qs = "?";
 70 			} else {
 71 				qs = "&";
 72 			}
 73 			for (var key in data) {
 74 				qs += key + "=" + data[key] + "&";
 75 			}
 76 			qs = qs.substr(0, qs.length - 1);
 77 		}
 78 		return qs;
 79 	}

 85 	/**
 86 		Creates the request and attaches all event handlers
 87 
 88 		@private
 89 		@param {String} url destination url
 90 		@param {Object} opts request options
 91 		@param {FormData|File|Blob} [data] request data
 92 		@param {function} [cb] callback function on success
 93 	*/
 94 	var	_request = function (url, opts, data, cb) {
 95 		var client = new XMLHttpRequest()
 96 			,	params = methods._mix(opts)
 97 			,	upload
 98 			,	requestData
 99 			,	callback;
100 		if (arguments.length === 3) { // data or callback?
101 			if (typeof arguments[2] === "function") {
102 				callback = arguments[2];
103 			} else {
104 				requestData = arguments[2];
105 			}
106 		}
107 		if (params.progress && typeof params.progress === "function") {
108 			upload = client.upload;
109 			if (upload) {
110 				upload.callback = params.prgress;
111 				upload.addEventListener("progress", methods._progress, false);
112 			}
113 		}
114 		methods._addHeaders(client, params.dType);
115 		client.open(params.type, url);
116 		client.xhr2data = params;
117 		client.callback = cb;
118 		client.addEventListener(
119 				"readystatechange"
120 			,	methods._stateChange
121 			,	false
122 		);
123 		client.addEventListener(
124 				"error"
125 			,	methods._handleRequestError
126 			,	false
127 		);
128 		client.send(requestData);
129 	}

131 	/**
132 		Mixes passed in options with the defaults
133 
134 		@private
135 	*/
136 	var _mix = function () {
137 		var ret = {}
138 			,	i = 0
139 			,	key;
140 		for (key in defaults) {
141 			ret[key] = defaults[key];
142 		}
143 		if (arguments.length && typeof arguments[0] !== "undefined") {
144 			for (; i < arguments.length; ++i) {
145 				for (key in arguments[i]) {
146 					ret[key] = arguments[i][key];
147 				}
148 			}
149 		}
150 		return ret; 
151 	}

153 	/**
154 		Event handler for readystatechange events
155 		Fires callback, if one was defined, with
156 		formatted data as required
157 
158 		@private
159 	*/
160 	var	_stateChange = function (ev) {
161 		var clientData = this.xhr2data;
162 		if (clientData) {
163 			if (this.readyState === 4) {
164 				if (clientData.dType === "json") {
165 					try {
166 						resBody = JSON.parse(this.responseText);
167 					} catch (ex) {
168 						// invalid json data, just pass response body
169 						resBody = this.responseText;
170 					}
171 				} else if (client.dType === "xml") {
172 					resBody = this.responseXML;
173 				} else {
174 					resBody = this.responseText;
175 				}
176 				if (this.callback) {
177 					this.callback.call(undefined, resBody);
178 				}
179 			}
180 		}
181 	}

201 	var	_addHeaders = function (client, resType) {
202 		var mime = "*/*, text/html"
